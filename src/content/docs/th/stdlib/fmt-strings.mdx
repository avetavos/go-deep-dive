## fmt — Formatted I/O

Package `fmt` คือจุดเริ่มต้นของ output เกือบทั้งหมดใน Go มีสามฟังก์ชันหลักที่ครอบคลุมกรณีการใช้งานส่วนใหญ่:

- `fmt.Println(args...)` — พิมพ์แต่ละ argument คั่นด้วยช่องว่าง แล้วขึ้นบรรทัดใหม่
- `fmt.Printf(format, args...)` — พิมพ์โดยใช้ format string ที่มี verb
- `fmt.Sprintf(format, args...)` — คืนค่า string ที่ฟอร์แมตแล้วแทนที่จะพิมพ์

### Format verbs

| Verb | ความหมาย |
|------|---------|
| `%v` | การแสดงผลแบบ default |
| `%+v` | Struct พร้อมชื่อ field |
| `%#v` | การแสดงผลแบบ Go syntax |
| `%T` | Type ของค่า |
| `%d` | เลขฐานสิบ (decimal integer) |
| `%s` | String (ไม่มีเครื่องหมายคำพูด) |
| `%q` | String ในเครื่องหมายคำพูดคู่พร้อม Go escaping |
| `%f` | ทศนิยม (floating-point decimal) |

```go
type Person struct { Name string; Age int }
p := Person{"Alice", 30}
fmt.Printf("%v\n",  p)   // {Alice 30}
fmt.Printf("%+v\n", p)   // {Name:Alice Age:30}
fmt.Printf("%#v\n", p)   // main.Person{Name:"Alice", Age:30}
fmt.Printf("%T\n",  p)   // main.Person
```

## strings — การจัดการข้อความ

Package `strings` ให้บริการฟังก์ชันสำหรับการค้นหา แยก รวม และแปลง string ใน Go นั้น string เป็น immutable UTF-8 byte sequence ดังนั้นฟังก์ชันเหล่านี้จะคืนค่า string ใหม่แทนที่จะแก้ไข in-place

```go
s := "hello, world"
strings.Contains(s, "world")         // true
strings.HasPrefix(s, "hello")        // true
strings.Split(s, ", ")               // ["hello" "world"]
strings.Join([]string{"a","b"}, "-") // "a-b"
strings.ReplaceAll(s, "l", "L")      // "heLLo, worLd"
strings.ToUpper(s)                   // "HELLO, WORLD"
strings.TrimSpace("  hi  ")          // "hi"
```

### strings.Builder — การต่อ string อย่างมีประสิทธิภาพ

เมื่อสร้าง string จากหลายส่วนในลูป ให้ใช้ `strings.Builder` แทน `+=` เพราะ `+=` จะสร้าง string ใหม่ทุกครั้งที่วนซ้ำ ในขณะที่ `Builder` เขียนลงใน buffer ที่ขยายตัวได้และ allocate string สุดท้ายเพียงครั้งเดียว

```go
var b strings.Builder
for i := 0; i < 5; i++ {
    fmt.Fprintf(&b, "item%d ", i)
}
result := strings.TrimSpace(b.String()) // "item0 item1 item2 item3 item4"
```

## strconv — การแปลงระหว่าง string และตัวเลข

`strconv` แปลงระหว่าง string และ numeric type ฟังก์ชันที่พบบ่อยที่สุด:

```go
n, err   := strconv.Atoi("42")          // string → int
s        := strconv.Itoa(42)            // int → string
f, err   := strconv.ParseFloat("3.14", 64) // string → float64
b, err   := strconv.ParseBool("true")   // string → bool
```

`Atoi` และ `ParseFloat` คืนค่า `(value, error)` เพราะ input อาจไม่ใช่ตัวเลขที่ถูกต้อง ควรตรวจสอบ error ก่อนใช้ค่าเสมอ

export const fmtStringsCode = `package main

import (
\t"fmt"
\t"strconv"
\t"strings"
)

type Person struct {
\tName string
\tAge  int
}

func main() {
\t// fmt verbs
\tp := Person{Name: "Alice", Age: 30}
\tfmt.Printf("%v\\n", p)
\tfmt.Printf("%+v\\n", p)
\tfmt.Printf("%#v\\n", p)
\tfmt.Printf("%T\\n", p)
\tfmt.Printf("%d %s %q\\n", 42, "hello", "world")

\t// Sprintf builds a string
\ts := fmt.Sprintf("Name: %s, Age: %d", p.Name, p.Age)
\tfmt.Println(s)

\t// strings package
\ttext := "  Go is great  "
\tfmt.Println(strings.TrimSpace(text))
\tfmt.Println(strings.Contains(text, "great"))
\tfmt.Println(strings.ToUpper(strings.TrimSpace(text)))

\twords := strings.Split("one,two,three", ",")
\tfmt.Println(words)
\tfmt.Println(strings.Join(words, " | "))
\tfmt.Println(strings.ReplaceAll("aabbcc", "b", "x"))

\t// strings.Builder — efficient concatenation
\tvar b strings.Builder
\tfor i := 0; i < 3; i++ {
\t\tfmt.Fprintf(&b, "item%d ", i)
\t}
\tfmt.Println(strings.TrimSpace(b.String()))

\t// strconv
\tn, err := strconv.Atoi("123")
\tif err == nil {
\t\tfmt.Println(n + 1)
\t}
\tfmt.Println(strconv.Itoa(456))
\tf, err := strconv.ParseFloat("3.14", 64)
\tif err == nil {
\t\tfmt.Printf("%.2f\\n", f)
\t}
}`;
